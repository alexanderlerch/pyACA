57 # initialization
58 f_min = 0
59 f_max = min(f_max, f_s / 2)

62
63 # compute center band frequencies
64 mel_min = ToolFreq2Mel(f_min)

70 f_u = f_mel[2:iNumFilters + 2]

74 # compute the transfer functions
